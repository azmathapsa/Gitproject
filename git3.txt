package assignment2;

public class Address {
	int doorNo;
	String street;
	String city;
	int pinCode;
	public Address(int doorNo, String street, String city, int pinCode) {
		this.doorNo = doorNo;
		this.street = street;
		this.city = city;
		this.pinCode = pinCode;
	}
	@Override
	public String toString() {
	return "Door Number " +doorNo+" Street "+street+" City "+city+" Pin Code "+pinCode;
	}
}
